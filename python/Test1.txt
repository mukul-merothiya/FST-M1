	
import pytest
 
# Additon test
def test_addition():
	
    # Initialize two numbers
	num1 = 10
	num2 = 15
    
	# Add them
	sum = num1 + num2
 
	# Assertion
	assert sum == 25
 
# Subtraction test
def test_subtraction():
  
	# Initialize two numbers
	num1 = 50
	num2 = 35
    
	# Subtract them
	diff = num1 - num2
 
	# Assertion
	assert diff == 15
